Git Commands 

git add FILE_NAME_GOES_HERE
  - add the file to the staged commits
git commit -m 'COMMIT MESSAGE GOES HERE'

git push REMOTE_NAME BRANCH_NAME

git pull REMOTE_NAME BRANCH_NAME


git status 
  -shows file changes 

git diff FILE_NAME
  -shows changes between file versions
